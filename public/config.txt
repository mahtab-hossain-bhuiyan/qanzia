# Redirect www → non-www (HTTPS)
server {
    listen 443 ssl;
    server_name www.qnaxia.com;
    ssl_certificate     /etc/letsencrypt/live/qnaxia.com/fullchain.pem;
    ssl_certificate_key /etc/letsencrypt/live/qnaxia.com/privkey.pem;
    return 301 https://qnaxia.com$request_uri;
}

# Main HTTPS server
server {
    listen 443 ssl;
    server_name qnaxia.com;

    ssl_certificate     /etc/letsencrypt/live/qnaxia.com/fullchain.pem;
    ssl_certificate_key /etc/letsencrypt/live/qnaxia.com/privkey.pem;
    ssl_protocols       TLSv1.2 TLSv1.3;
    ssl_prefer_server_ciphers on;

    gzip on;
    gzip_types text/plain text/css application/javascript application/json;
    gzip_min_length 1000;

    root /var/www/qnazia/public;

    # API requests → Node.js (MUST be before location /)
    location /api/ {
        proxy_pass http://127.0.0.1:3000;
        proxy_http_version 1.1;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        client_max_body_size 10M;

        add_header X-Robots-Tag "noindex" always;
        proxy_set_header CF-Access-Client-Id "bypass";
    }

    # Sitemap → Node.js
    location = /sitemap.xml {
        proxy_pass http://127.0.0.1:3000;
        proxy_set_header Host $host;
    }

    # Everything else → serve index.html (SPA)
    location / {
        try_files $uri $uri/ /index.html;
    }
}
server {
    server_name api.qnaxia.com;

    location / {
        proxy_pass http://127.0.0.1:3000;
        proxy_http_version 1.1;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        client_max_body_size 10M;

        # CORS — allow your main domain to call this
    }

    listen 443 ssl; # managed by Certbot
    ssl_certificate /etc/letsencrypt/live/api.qnaxia.com/fullchain.pem; # managed by Certbot
    ssl_certificate_key /etc/letsencrypt/live/api.qnaxia.com/privkey.pem; # managed by Certbot
    include /etc/letsencrypt/options-ssl-nginx.conf; # managed by Certbot
    ssl_dhparam /etc/letsencrypt/ssl-dhparams.pem; # managed by Certbot

}

server {
    if ($host = api.qnaxia.com) {
